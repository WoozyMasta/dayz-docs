343 // Calculates and return temperature of environment
344 protected float GetEnvironmentTemperature()
345 {
346 float temperature;
347 temperature = g_Game.GetMission().GetWorldData().GetBaseEnvTemperature();
348 temperature += Math.AbsFloat(temperature * m_Clouds * GameConstants.ENVIRO_CLOUDS_TEMP_EFFECT);
349
350 if (IsWaterContact())
351 temperature -= Math.AbsFloat(temperature * GameConstants.ENVIRO_WATER_TEMPERATURE_COEF);
352
353 if (IsInsideBuilding() || m_IsUnderRoofBuilding)
354 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_INSIDE_COEF);
355 else if (IsInsideVehicle())
356 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_INSIDE_VEHICLE_COEF);
357 else if (IsUnderRoof() && !m_IsUnderRoofBuilding)
358 {
359 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_UNDERROOF_COEF);
360 temperature -= GameConstants.ENVIRO_TEMPERATURE_WIND_COEF * GetWindModifierPerSurface() * m_Wind;
361 }
362 else
363 {
364 temperature -= GameConstants.ENVIRO_TEMPERATURE_WIND_COEF * GetWindModifierPerSurface() * m_Wind;
365 temperature -= Math.AbsFloat(temperature * m_Fog * GameConstants.ENVIRO_FOG_TEMP_EFFECT);
366 temperature -= GetTemperatureHeightCorrection();
367 }
368
369 // incorporate temperature from temperature sources (buffer)
370 if (Math.AbsFloat(m_UTSAverageTemperature) > 0.001)
371 temperature += m_UTSAverageTemperature;
372
373 return temperature;
374 }

376 // Calculats wet/drying delta based on player's location and weather
377 float GetWetDelta()
378 {
379 float wetDelta = 0;
380 if (IsWaterContact())
381 {
383 if (m_WaterLevel >= WATER_LEVEL_HIGH)
384 wetDelta = 1;
385 else if (m_WaterLevel >= WATER_LEVEL_MID && m_WaterLevel < WATER_LEVEL_HIGH)
386 wetDelta = 0.66;
387 else if (m_WaterLevel >= WATER_LEVEL_LOW && m_WaterLevel < WATER_LEVEL_MID)
388 wetDelta = 0.66;
389 else if (m_WaterLevel >= WATER_LEVEL_NONE && m_WaterLevel < WATER_LEVEL_LOW)
390 wetDelta = 0.33;
391 }
392 else if (IsRaining() && !IsInsideBuilding() && !IsUnderRoof() && !IsInsideVehicle())
393 {
395 wetDelta = GameConstants.ENVIRO_WET_INCREMENT * GameConstants.ENVIRO_TICKS_TO_WETNESS_CALCULATION * (m_Rain) * (1 + (GameConstants.ENVIRO_WIND_EFFECT * m_Wind));
396 }
397 else
398 {
400 float tempEffect = Math.Max(m_PlayerHeat + GetEnvironmentTemperature(), 1.0);
401
402 float weatherEffect = ((1 - (m_Fog * GameConstants.ENVIRO_FOG_DRY_EFFECT))) * (1 - (m_Clouds * GameConstants.ENVIRO_CLOUD_DRY_EFFECT));
403 if (weatherEffect <= 0)
404 weatherEffect = 1.0;
405
406 wetDelta = -(GameConstants.ENVIRO_DRY_INCREMENT * weatherEffect * tempEffect);
407 if (!IsInsideBuilding())
408 wetDelta *= 1 + (GameConstants.ENVIRO_WIND_EFFECT * m_Wind);
409 }
410
411 return wetDelta;
412 }

741
742 BodyPartHeatProperties(m_HeadParts, GameConstants.ENVIRO_HEATCOMFORT_HEADPARTS_WEIGHT, hcHead, hHead);
743 BodyPartHeatProperties(m_BodyParts, GameConstants.ENVIRO_HEATCOMFORT_BODYPARTS_WEIGHT, hcBody, hBody);
744 BodyPartHeatProperties(m_FeetParts, GameConstants.ENVIRO_HEATCOMFORT_FEETPARTS_WEIGHT, hcFeet, hFeet);
745
746 heatComfortAvg = (hcHead + hcBody + hcFeet) / 3;
747 heatAvg = (hHead + hBody + hFeet) / 3;
748 heatAvg = heatAvg * GameConstants.ENVIRO_ITEM_HEAT_TRANSFER_COEF;
749
750 // heat buffer
751 float applicableHB = 0.0;
752 if (m_UTSAverageTemperature < 0.001)
753 {
754 applicableHB = m_Player.GetStatHeatBuffer().Get() / 30.0;
755 if (applicableHB > 0.0)
756 {
757 if (m_HeatBufferTimer > 1.0)
758 m_Player.GetStatHeatBuffer().Add(Math.Min(EnvTempToCoef(m_EnvironmentTemperature), -0.1) * GameConstants.ENVIRO_PLAYER_HEATBUFFER_DECREASE);
759 else
760 m_HeatBufferTimer += GameConstants.ENVIRO_PLAYER_HEATBUFFER_TICK;
761 }
762 else
763 m_HeatBufferTimer = 0.0;
764 }
765 else
766 {
767 applicableHB = m_Player.GetStatHeatBuffer().Get() / 30.0;
768 if (m_HeatComfort > PlayerConstants.THRESHOLD_HEAT_COMFORT_MINUS_WARNING)
769 {
770 m_Player.GetStatHeatBuffer().Add(GameConstants.ENVIRO_PLAYER_HEATBUFFER_INCREASE);
771 m_HeatBufferTimer = 0.0;
772 }
773 else
774 m_HeatBufferTimer = 0.0;
775 }
776
777 m_HeatComfort = (heatComfortAvg + heatAvg + (GetPlayerHeat() / 100)) + EnvTempToCoef(m_EnvironmentTemperature);
778 if ((m_HeatComfort + applicableHB) < (PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01))
779 m_HeatComfort += applicableHB;
780 else
781 {
782 if (m_HeatComfort <= (PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01))
783 m_HeatComfort = PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01;
784 }
785
786 m_HeatComfort = Math.Clamp(m_HeatComfort, m_Player.GetStatHeatComfort().GetMin(), m_Player.GetStatHeatComfort().GetMax());
787
788 m_Player.GetStatHeatComfort().Set(m_HeatComfort);
789 }

806 protected void BodyPartHeatProperties(array<int> pBodyPartIds, float pCoef, out float pHeatComfort, out float pHeat)
807 {
808 int attCount;
809
810 EntityAI attachment;
811 ItemBase item;
812
813 pHeatComfort = -1;
814 attCount = m_Player.GetInventory().AttachmentCount();
815
816 for (int attIdx = 0; attIdx < attCount; attIdx++)
817 {
818 attachment = m_Player.GetInventory().GetAttachmentFromIndex(attIdx);
819 if (attachment.IsClothing())
820 {
821 item = ItemBase.Cast(attachment);
822 int attachmentSlot = attachment.GetInventory().GetSlotId(0);
823
825 for (int i = 0; i < pBodyPartIds.Count(); i++)
826 {
827 if (attachmentSlot == pBodyPartIds.Get(i))
828 {
829 float heatIsoMult = 1.0;
830 if (attachmentSlot == InventorySlots.VEST)
831 heatIsoMult = GameConstants.ENVIRO_HEATISOLATION_VEST_WEIGHT;
832
833 if (attachmentSlot == InventorySlots.BACK)
834 heatIsoMult = GameConstants.ENVIRO_HEATISOLATION_BACK_WEIGHT;
835
836 pHeatComfort += heatIsoMult * MiscGameplayFunctions.GetCurrentItemHeatIsolation(item);
837
838 // go through any attachments and cargo (only current level, ignore nested containers - they isolate)
839 int inAttCount = item.GetInventory().AttachmentCount();
840 if (inAttCount > 0)
841 {
842 for (int inAttIdx = 0; inAttIdx < inAttCount; inAttIdx++)
843 {
844 EntityAI inAttachment = item.GetInventory().GetAttachmentFromIndex(inAttIdx);
845 ItemBase itemAtt = ItemBase.Cast(inAttachment);
846 if (itemAtt)
847 pHeat += itemAtt.GetTemperature();
848 }
849 }
850 if (item.GetInventory().GetCargo())
851 {
852 int inItemCount = item.GetInventory().GetCargo().GetItemCount();
853
854 for (int j = 0; j < inItemCount; j++)
855 {
856 ItemBase inItem;
857 if (Class.CastTo(inItem, item.GetInventory().GetCargo().GetItem(j)))
858 pHeat += inItem.GetTemperature();
859 }
860 }
861 }
862 }
863 }
864 }
865
866 pHeatComfort = (pHeatComfort / pBodyPartIds.Count()) * pCoef;
867 pHeat = (pHeat / pBodyPartIds.Count()) * pCoef;

900 return;

904

906 foreach (UTemperatureSource tempSource : m_UTemperatureSources)

911
912 if (max > 0 && min < 0)
913 {
915 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add((max + min) * 0.5);
916 }
917 else
918 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add(max);

920 m_HasTemperatureSources = true;

931 distance = Math.Max(distance, 0.1); //min distance cannot be 0 (division by zero)

935
937 if (distance > uts.GetFullRange())
938 {
939 float distFactor = 1 - (distance / uts.GetMaxRange());
940 distFactor = Math.Max(distFactor, 0.0);
941 temperature = uts.GetTemperature() * distFactor;
942 //temperature = Math.Clamp(temperature, uts.GetTemperatureMin(), uts.GetTemperatureMax());
943 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::distFactor: %1", distFactor), "Environment");
944 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::temperatureXX: %1", temperature), "Environment");
945 }
946 else
947 temperature = uts.GetTemperature();
948
949 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::temperature: %1", temperature), "Environment");
950
951 return temperature;
