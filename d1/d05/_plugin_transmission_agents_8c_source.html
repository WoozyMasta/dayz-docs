189 float TransmitAgentsEx(EntityAI source, EntityAI target, int pathway, int dose_size = 1000, int agents = 0)
190 {
191 //Debug.Log("Transmitting agents for source: " +source.ToString()+", target: " +target.ToString(),"Agents");
192 int sourceAgents = agents;
193 int targetAgents;
194 if (!sourceAgents && source) sourceAgents = source.GetAgents(); //do not set sourceAgents again if already set
195 if (target) targetAgents = target.GetAgents();
196 float count = 0;
197
198 switch (pathway)
199 {
200 case AGT_INV_OUT: //item leaving inventory
201 break;
202
203 case AGT_INV_IN: //item entering inventory
204 break;
205
206 case AGT_UACTION_TOUCH: //player touched the item
207 //InjectAgents( source, targetAgents ,GetProtectionLevel(DEF_BIOLOGICAL,InventorySlots.GLOVES, player) );
208 break;
209
210 case AGT_WATER_POND:
211 //target.InsertAgent(eAgents.CHOLERA, dose_size);
212 InjectAgentsWithPlayer(target, eAgents.CHOLERA, 0, 1, InjectTypes.ITEM_TO_PLAYER);
213 break;
214
215 case AGT_UACTION_CONSUME:
216 //InjectAgentsWithPlayer( target, sourceAgents , 0, dose_size, InjectTypes.ITEM_TO_PLAYER );
217 InjectAgentsWithPlayer(source, targetAgents, 0, 1, InjectTypes.PLAYER_TO_ITEM);
218 break;
219
220 case AGT_UACTION_TO_PLAYER: //user action of a consumption, only from item to player
221 InjectAgentsWithPlayerCount(target, sourceAgents, 0, dose_size, InjectTypes.ITEM_TO_PLAYER);
222 break;
223
224 case AGT_UACTION_TO_ITEM: //to transfer from the player to the consumed item
225 InjectAgentsWithPlayer(target, sourceAgents, 0, 1, InjectTypes.PLAYER_TO_ITEM);
226 break;
227
228 case AGT_TRANSFER_COPY: //transferring liquid
229 InjectAgentsWithoutPlayer(target, sourceAgents);
230 break;
231
232 case AGT_ITEM_TO_FLESH: //transferring liquid
233 InjectAgentsWithPlayer(target, sourceAgents, 0, 1, InjectTypes.ITEM_TO_PLAYER);
234 break;
235
236 case AGT_AIRBOURNE_BIOLOGICAL:
237 float prot_level_mask_target = GetProtectionLevel(DEF_BIOLOGICAL, InventorySlots.MASK, Man.Cast(target));
238 float prot_level_mask_source = GetProtectionLevel(DEF_BIOLOGICAL, InventorySlots.MASK, Man.Cast(source));
239 float prot_level_headgear_target = GetProtectionLevel(DEF_BIOLOGICAL, InventorySlots.HEADGEAR, Man.Cast(target));
240 float prot_level_headgear_source = GetProtectionLevel(DEF_BIOLOGICAL, InventorySlots.HEADGEAR, Man.Cast(source));
241 float prot_level_target = Math.Max(prot_level_mask_target, prot_level_headgear_target);//find the bigger of the 2, TODO: should be improved
242 float prot_level_source = Math.Max(prot_level_mask_source, prot_level_headgear_source);//find the bigger of the 2, TODO: should be improved
243 float prot_level_combined = 1 - (1 - prot_level_target) * (1 - prot_level_source);
244 InjectAgentsWithPlayer(target, sourceAgents, prot_level_combined, 1, InjectTypes.PLAYER_AIR_PLAYER);
245 break;
246 case AGT_AIRBOURNE_CHEMICAL:
247 float prot_level_mask_target2 = GetProtectionLevel(DEF_CHEMICAL, InventorySlots.MASK, Man.Cast(target));
248
249 count = InjectAgentWithPlayerDose(target, agents, prot_level_mask_target2, dose_size, InjectTypes.PLAYER_AIR_PLAYER);
250 break;
251 }
252 return count;
253 }

272 {
273 if (target && (agents != 0) && target.IsEntityAI())
