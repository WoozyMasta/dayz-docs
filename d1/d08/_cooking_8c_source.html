35 {
36 Edible_Base item_to_cook = Edible_Base.Cast(pItem);
37
39 pStateFlags = new Param2<bool, bool>(false, false);
40
41 if (item_to_cook && item_to_cook.CanBeCooked())
42 {
44 item_to_cook.MakeSoundsOnClient(true, pCookingMethod.param1);
45
47 UpdateCookingState(item_to_cook, pCookingMethod.param1, cookingEquip, pCookingMethod.param2);

149 protected int UpdateCookingState(Edible_Base item_to_cook, CookingMethodType cooking_method, ItemBase cooking_equipment, float cooking_time_coef)
150 {
151 //food properties
152 float food_temperature = item_to_cook.GetTemperature();
153
154 //{min_temperature, time_to_cook, max_temperature (optional)}
155 //get next stage name - if next stage is not defined, next stage name will be empty "" and no cooking properties (food_min_temp, food_time_to_cook, food_max_temp) will be set
156 FoodStageType new_stage_type = item_to_cook.GetNextFoodStageType(cooking_method);
157
158 float food_min_temp = 0;
159 float food_time_to_cook = 0;
160 float food_max_temp = -1;
161
162 //Set next stage cooking properties if next stage possible
163 if (item_to_cook.CanChangeToNewStage(cooking_method)) // new_stage_type != NONE
164 {
165 array<float> next_stage_cooking_properties = new array<float>();
166 next_stage_cooking_properties = FoodStage.GetAllCookingPropertiesForStage(new_stage_type, null, item_to_cook.GetType());
167
168 food_min_temp = next_stage_cooking_properties.Get(eCookingPropertyIndices.MIN_TEMP);
169 food_time_to_cook = next_stage_cooking_properties.Get(eCookingPropertyIndices.COOK_TIME);
170 // The last element is optional and might not exist
171 if (next_stage_cooking_properties.Count() > 2)
172 food_max_temp = next_stage_cooking_properties.Get(eCookingPropertyIndices.MAX_TEMP);
173 }
174
175 //add temperature
176 AddTemperatureToItem(item_to_cook, cooking_equipment, food_min_temp);
177
178 //add cooking time if the food can be cooked by this method
179 if (food_min_temp > 0 && food_temperature >= food_min_temp)
180 {
181 float new_cooking_time = item_to_cook.GetCookingTime() + COOKING_FOOD_TIME_INC_VALUE * cooking_time_coef;
182 item_to_cook.SetCookingTime(new_cooking_time);
183
184 //progress to next stage
185 if (item_to_cook.GetCookingTime() >= food_time_to_cook)
186 {
187 //if max temp is defined check next food stage
188 if (food_max_temp >= 0)
189 {
190 if (food_temperature > food_max_temp && item_to_cook.GetFoodStageType() != FoodStageType.BURNED)
191 new_stage_type = FoodStageType.BURNED;
192 }
193
195 item_to_cook.ChangeFoodStage(new_stage_type);
197 item_to_cook.RemoveAllAgentsExcept(eAgents.BRAIN);
198
199 if (cooking_equipment)
200 {
201 if (cooking_method == CookingMethodType.BAKING)
202 {
203 ItemBase lard = GetItemTypeFromCargo(COOKING_INGREDIENT_LARD, cooking_equipment);
204 if (lard)
205 {
206 //decrease lard quantity
207 float lardQuantity = lard.GetQuantity() - COOKING_LARD_DECREASE_COEF;
208 lardQuantity = Math.Clamp(lardQuantity, 0, lard.GetQuantityMax());
209 lard.SetQuantity(lardQuantity);
210 }
211 else
212 {
214 DecreaseCookedItemQuantity(item_to_cook, COOKING_FOOD_QUANTITY_DECREASE_AMOUNT_NONE);
215 }
216 }
217 }
218 else
219 {
221 DecreaseCookedItemQuantity(item_to_cook, COOKING_FOOD_QUANTITY_DECREASE_AMOUNT_NONE);
222 }
223
224 //reset cooking time
225 item_to_cook.SetCookingTime(0);
226
227 return 1;
228 }
229 }
230
231 return 0;

408 protected Param2<CookingMethodType, float> GetCookingMethodWithTimeOverride(ItemBase cooking_equipment)
409 {
410 Param2<CookingMethodType, float> val = new Param2<CookingMethodType, float>(CookingMethodType.NONE, TIME_WITH_SUPPORT_MATERIAL_COEF);
411
412 switch (cooking_equipment.Type())
413 {
414 case COOKING_EQUIPMENT_POT:
415 case COOKING_EQUIPMENT_CAULDRON:
416 case COOKING_EQUIPMENT_FRYINGPAN:
417 if (cooking_equipment.GetQuantity() > 0)
418 {
419 if (cooking_equipment.GetLiquidType() == LIQUID_GASOLINE)
420 {
422 val = new Param2<CookingMethodType, float>(CookingMethodType.DRYING, TIME_WITHOUT_SUPPORT_MATERIAL_COEF);
423 break;
424 }
425
426 val = new Param2<CookingMethodType, float>(CookingMethodType.BOILING, TIME_WITH_SUPPORT_MATERIAL_COEF);
427 break;
428 }
429
430 if (GetItemTypeFromCargo(COOKING_INGREDIENT_LARD, cooking_equipment))
431 {
432 //has lard in cargo, slower process
433 val = new Param2<CookingMethodType, float>(CookingMethodType.BAKING, TIME_WITH_SUPPORT_MATERIAL_COEF);
434 break;
435 }
436
437 if (cooking_equipment.GetInventory().GetCargo().GetItemCount() > 0)
438 {
439 val = new Param2<CookingMethodType, float>(CookingMethodType.BAKING, TIME_WITHOUT_SUPPORT_MATERIAL_COEF);
440 break;
441 }
442
443 val = new Param2<CookingMethodType, float>(CookingMethodType.NONE, TIME_WITHOUT_SUPPORT_MATERIAL_COEF);
444 break;
445
446 default:
447 val = new Param2<CookingMethodType, float>(CookingMethodType.BAKING, TIME_WITHOUT_SUPPORT_MATERIAL_COEF);
448 break;
449 }
450
451 return val;
