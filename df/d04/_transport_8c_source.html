215 bool IsAreaAtDoorFree(int currentSeat, float maxAllowedObjHeight, inout vector extents, out vector transform[4])
216 {
217 GetTransform(transform);
218
219 vector crewPos;
220 vector crewDir;
221 CrewEntry(currentSeat, crewPos, crewDir);
222
223 vector entry[4];
224 entry[2] = crewDir;
225 entry[0] = vector.Up * crewDir;
226 entry[1] = entry[2] * entry[0];
227 entry[3] = crewPos;
228
229 Math3D.MatrixMultiply4(transform, entry, transform);
230
231 vector position = transform[3];
232 vector orientation = Math3D.MatrixToAngles(transform);
233
234 position[1] = position[1] + maxAllowedObjHeight + (extents[1] * 0.5);
235
236 array<Object> excluded = new array<Object>;
237 array<Object> collided = new array<Object>;
238
239 excluded.Insert(this);
240
241 GetGame().IsBoxColliding(position, orientation, extents, excluded, collided);
242
243 orientation.RotationMatrixFromAngles(transform);
244 transform[3] = position;
245
246 foreach (Object o : collided)
247 {
248 EntityAI e = EntityAI.Cast(o);
249 if (IsIgnoredObject(o))
250 continue;
251
252 vector minmax[2];
253 if (o.GetCollisionBox(minmax))
254 return false;
255 }
