101 void PluginsInit()
102 {
103 int i = 0;
104 int regCount = m_PluginRegister.Count();
105
106 array<PluginBase> pluginPtrs = {};
107 pluginPtrs.Reserve(regCount);
108
109 foreach (typename pluginType : m_PluginRegister)
110 {
111 PluginBase moduleExist = GetPluginByType(pluginType);
112 if (moduleExist)
113 m_PluginsPtrs.Remove(pluginType);
114
115 PluginBase moduleNew = PluginBase.Cast(pluginType.Spawn());
116 m_PluginsPtrs.Set(pluginType, moduleNew);
117 pluginPtrs.Insert(moduleNew);
118 }
119
120 foreach (PluginBase plugin : pluginPtrs)
121 {
122 if (plugin)
123 plugin.OnInit();
124 }
125 }
