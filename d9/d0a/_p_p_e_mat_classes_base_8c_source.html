224 void SetFinalParameterValue(int parameter_idx)
225 {
226 int var_type = GetParameterCommandData(parameter_idx).GetParameterVarType();
227 Param values = GetParameterCommandData(parameter_idx).GetCurrentValues();
228
229 switch (var_type)
230 {
231 case PPEConstants.VAR_TYPE_BOOL:
232 bool value_var_bool = Param1<bool>.Cast(values).param1;
233
234 m_Material.SetParamByIndex(parameter_idx, value_var_bool);
235 //Print("DebugValues | PPEConstants.VAR_TYPE_BOOL | bool val: " + value_var_bool);
236 break;
237
238 case PPEConstants.VAR_TYPE_INT:
239 int value_var_int = Param1<int>.Cast(values).param1;
240
241 m_Material.SetParamByIndex(parameter_idx, value_var_int);
242 //Print("DebugValues | PPEConstants.VAR_TYPE_BOOL | bool val: " + value_var_bool);
243 break;
244
245 case PPEConstants.VAR_TYPE_FLOAT:
246 float value_var_float = Param1<float>.Cast(values).param1;
247
248 m_Material.SetParamByIndex(parameter_idx, value_var_float);
249 //Print("DebugValues | PPEConstants.VAR_TYPE_FLOAT | float val: " + value_var_float);
250 break;
251
252 case PPEConstants.VAR_TYPE_COLOR:
253 float color[4] = {0, 0, 0, 0};
254 color[0] = Param4<float, float, float, float>.Cast(values).param1;
255 color[1] = Param4<float, float, float, float>.Cast(values).param2;
256 color[2] = Param4<float, float, float, float>.Cast(values).param3;
257 color[3] = Param4<float, float, float, float>.Cast(values).param4;
258
259 m_Material.SetParamByIndex(parameter_idx, color);
260 //Print("DebugValues | PPEConstants.VAR_TYPE_COLOR | color val:: " + color[0] + " " + color[1] + " " + color[2] + " " + color[3]);
261 break;
