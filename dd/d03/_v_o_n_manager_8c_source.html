167 override void OnEvent(EventType eventTypeId, Param params)
168 {
169 Mission mission = GetGame().GetMission();
170 switch (eventTypeId)
171 {
172 case VONUserStartedTransmittingAudioEventTypeID:
173 {
174 // only handle this if we are in Voice Activation mode, so ignore if in PTT mode
175 if (m_VoNToggled)
176 {

178 ShowVoiceNotification(GetGame().GetVoiceLevel(), false);
179 }
180 break;
181 }
182
183 case VONUserStoppedTransmittingAudioEventTypeID:
184 {
185 // only handle this if we are in Voice Activation mode, so ignore if in PTT mode
186 if (m_VoNToggled)
187 {
188 if (!VONManager.IsVoiceThresholdMinimum())
189 HideVoiceNotification();
190 }
191 break;
192 }
193
194 case VONStateEventTypeID:
195 {
196 if (!mission)
197 break;
198
199 VONStateEventParams vonStateParams = VONStateEventParams.Cast(params);
200 mission.SetVoNActive(vonStateParams.param1);
201 m_VoNToggled = vonStateParams.param2;
202
203 UpdateVoiceIcon();
204
205 m_OnVonStateEvent.Invoke();
206 break;
207 }
208
209 case PartyChatStatusChangedEventTypeID:
210 {
211 m_OnPartyChatChangedEvent.Invoke();
212 break;
213 }
214
215 case VONStartSpeakingEventTypeID:
216 {
217 VONStartSpeakingEventParams vonStartParams;
218 if (Class.CastTo(vonStartParams, params))
219 GetDayZGame().AddVoiceNotification(vonStartParams);
220 break;
221 }
222

224 {
225 VONStopSpeakingEventParams vonStopParams;
