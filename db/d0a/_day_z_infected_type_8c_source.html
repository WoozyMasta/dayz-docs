169 protected float ComputeAttackUtility(DayZInfectedAttackType pAttackType, float pTargetDistance, int pPitch, float pProbability)
170 {
171 // pitch
172 if (pAttackType.m_Pitch != pPitch)
173 return 0;
174
175 // distance
176 float distDiff = pAttackType.m_Distance - pTargetDistance;
177 if (distDiff < 0)
178 return 0;
179
180 float distDiffFrac = distDiff / 10;
181 float utilityDistance = (1 - distDiffFrac) * 100; // distance is most important
182
183 // probability

185 if (probDiff < 0)
186 return 0;
187
188 float utilityProbability = (1 - probDiff) * 10; // distance is most important
189
190 // total
191 float util = utilityDistance + utilityProbability;
192 //Print("Attack Utility " + util);
