418 SymptomBase QueueUpPrimarySymptom(int symptom_id, int uid = -1)
419 {
420 SymptomBase Symptom;
421 if ((GetSymptomCount(symptom_id) >= GetSymptomMaxCount(symptom_id)) && GetSymptomMaxCount(symptom_id) != -1)
422 return null;
423 for (int i = 0; i < m_SymptomQueuePrimary.Count(); i++)
424 {
425 if (m_SymptomQueuePrimary.Get(i).CanBeInterupted() && ComparePriority(GetSymptomPriority(symptom_id), m_SymptomQueuePrimary.Get(i).GetPriority()) == 1)
426 {
427 Symptom = SpawnSymptom(symptom_id, uid);
428 m_SymptomQueuePrimary.InsertAt(Symptom, i);
429
430 if (m_SymptomQueuePrimary.Count() > MAX_QUEUE_SIZE)
431 {
432 m_SymptomQueuePrimary.Get(MAX_QUEUE_SIZE).RequestDestroy();// no need to remove from the array, that's done via Symptom callback on destruct
433 }
434 break;
435 }
436 }
437 if (!Symptom && m_SymptomQueuePrimary.Count() < MAX_QUEUE_SIZE)
438 {
439 Symptom = SpawnSymptom(symptom_id, uid);
440 m_SymptomQueuePrimary.Insert(Symptom);
441 }
442#ifdef DIAG_DEVELOPER
443 SendServerDebugToClient();
444#endif
445 return Symptom;
446 }
