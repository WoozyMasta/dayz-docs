563 {
564 int screen_w, screen_h;

566 GetScreenSize(screen_w, screen_h);

671 static int GetChosenCombinationFlag(EntityAI selectedEntity, EntityAI targetEntity, int relevantFlags, out InventoryLocation dst = null)
672 {
673 PlayerBase player = PlayerBase.Cast(GetGame().GetPlayer());
674
675 if (!selectedEntity || !targetEntity)
676 return InventoryCombinationFlags.NONE;
677
678 ItemBase selectedItem = ItemBase.Cast(selectedEntity);
679 ItemBase targetItem = ItemBase.Cast(targetEntity);
680
681 //EntityAI itemInHands = m_player.GetHumanInventory().GetEntityInHands();
682 ActionManagerClient amc = ActionManagerClient.Cast(player.GetActionManager());
683
684 if (relevantFlags & InventoryCombinationFlags.PERFORM_ACTION)
685 {
686 if (amc.CanPerformActionFromInventory(targetItem, selectedItem))
687 return InventoryCombinationFlags.PERFORM_ACTION;
688 }
689
690 if (relevantFlags & InventoryCombinationFlags.SET_ACTION)
691 {
692 if (amc.CanSetActionFromInventory(targetItem, selectedItem))
693 return InventoryCombinationFlags.SET_ACTION;
694 }
695
696 if (relevantFlags & InventoryCombinationFlags.COMBINE_QUANTITY2)
697 {
698 if (targetEntity.CanBeCombined(selectedEntity))
699 return InventoryCombinationFlags.COMBINE_QUANTITY2;
700 }
701
702 if (relevantFlags & InventoryCombinationFlags.ADD_AS_ATTACHMENT)
703 {
704 if (targetEntity.GetInventory().CanAddAttachment(selectedEntity))
705 return InventoryCombinationFlags.ADD_AS_ATTACHMENT;
706 }
707
708 if (relevantFlags & InventoryCombinationFlags.ADD_AS_CARGO)
709 {
710 if (!targetEntity.GetInventory().HasEntityInInventory(selectedEntity) && targetEntity.GetInventory().CanAddEntityInCargo(selectedEntity, selectedEntity.GetInventory().GetFlipCargo()))
711 return InventoryCombinationFlags.ADD_AS_CARGO;
712 }
713
714 if (relevantFlags & InventoryCombinationFlags.SWAP_MAGAZINE)
715 {
716 Magazine mag = Magazine.Cast(targetEntity);
717 Weapon_Base wpn = Weapon_Base.Cast(targetEntity.GetHierarchyParent());
718 if (wpn && mag)
719 {
720 if (player.GetWeaponManager().CanSwapMagazine(wpn, Magazine.Cast(selectedEntity)))
721 return InventoryCombinationFlags.SWAP_MAGAZINE;
722 }
723 }
724
725 if (relevantFlags & InventoryCombinationFlags.SWAP)
726 {
727 if (GameInventory.CanSwapEntitiesEx(selectedEntity, targetEntity))
728 return InventoryCombinationFlags.SWAP;
729 }
730
731 if (relevantFlags & InventoryCombinationFlags.FSWAP)
732 {
733 if (GameInventory.CanForceSwapEntitiesEx(selectedEntity, null, targetEntity, dst))
734 return InventoryCombinationFlags.FSWAP;
735 }
736
737 return InventoryCombinationFlags.NONE;
738 }

740 static int GetCombinationFlags(EntityAI entity1, EntityAI entity2)
741 {
742 int flags = 0;
743 PlayerBase m_player = PlayerBase.Cast(GetGame().GetPlayer());
744
745 if (!entity1 || !entity2)
746 return flags;
747
748 if (entity1.IsInherited(ItemBase) && entity2.IsInherited(ItemBase))
749 {
750 ItemBase ent1 = ItemBase.Cast(entity1);
751 if (ent1.CanBeCombined(ItemBase.Cast(entity2))) flags = flags | InventoryCombinationFlags.COMBINE_QUANTITY2;
752 }
753
754 if (entity1.GetInventory().CanAddAttachment(entity2))
755 {
756 if (!entity1.IsInherited(ZombieBase) && !entity1.IsInherited(Car) && !entity2.IsInherited(ZombieBase) && !entity2.IsInherited(Car))
757 flags = flags | InventoryCombinationFlags.ADD_AS_ATTACHMENT;
758 }
759 if (!entity1.GetInventory().HasEntityInInventory(entity2) && entity1.GetInventory().CanAddEntityInCargo(entity2, entity2.GetInventory().GetFlipCargo())) flags = flags | InventoryCombinationFlags.ADD_AS_CARGO;
760
761 if (entity1 == m_player.GetHumanInventory().GetEntityInHands() || entity2 == m_player.GetHumanInventory().GetEntityInHands())
762 {
763 ActionManagerClient amc;
764 Class.CastTo(amc, m_player.GetActionManager());
765 if (entity1 == m_player.GetHumanInventory().GetEntityInHands())
766 {
767 if (amc.CanPerformActionFromInventory(ItemBase.Cast(entity1), ItemBase.Cast(entity2)))
768 flags = flags | InventoryCombinationFlags.PERFORM_ACTION;
769 else if (amc.CanSetActionFromInventory(ItemBase.Cast(entity1), ItemBase.Cast(entity2)))
770 flags = flags | InventoryCombinationFlags.SET_ACTION;
771 }
772 else
773 {
774 if (amc.CanPerformActionFromInventory(ItemBase.Cast(entity2), ItemBase.Cast(entity1)))
775 flags = flags | InventoryCombinationFlags.PERFORM_ACTION;
776 else if (amc.CanSetActionFromInventory(ItemBase.Cast(entity2), ItemBase.Cast(entity1)))
777 flags = flags | InventoryCombinationFlags.SET_ACTION;
778 }
779 }
780 return flags;
781 }
