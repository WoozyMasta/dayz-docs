145 bool SoundPlayEx(out SoundParams params)
146 {
147 super.Start();
148
149 if (m_SoundSetName != "")
150 {
151 vector position = GetCurrentLocalPosition();
152
153 if (SoundLoadEx(params))
154 {
155 if (m_SetEnvVariables && m_SoundParams)
156 {
157 m_SoundObjectBuilder.AddEnvSoundVariables(GetPosition());
158 m_SoundObject = m_SoundObjectBuilder.BuildSoundObject();
159 m_SoundObject.SetKind(m_SoundWaveKind);
160 m_SoundObject.SetParent(m_ParentObject);

162
163 if (m_SoundObject)
164 {
165 SetCurrentLocalPosition(position, false);
166 m_SoundWaveObject = GetGame().GetSoundScene().Play3D(m_SoundObject, m_SoundObjectBuilder);
167 if (!m_SoundWaveObject)
168 return false;
169
170 // Wait for header to be loaded before asking for its length, else we block the main thread
171 if (m_SoundWaveObject.IsHeaderLoaded())
172 ValidateSoundWave();
173 else
174 m_SoundWaveObject.GetEvents().Event_OnSoundWaveHeaderLoaded.Insert(ValidateSoundWave);
175
176 return true;
177 }
178 else
179 SoundError("m_SoundObject is null.");
180 }

427 override void Event_OnFrameUpdate(float time_delta)
428 {
429 if (IsSoundPlaying())
430 {
431 if (m_SoundDoppler != -1)
432 m_SoundWaveObject.SetDoppler(m_SoundDoppler);
433 // FadeIn
434 if (m_SoundWaveStarting)
435 {
436 if (m_SoundFadeInDuration > 0)
437 {
438 SetSoundVolume(GetSoundVolume() + (time_delta / m_SoundFadeInDuration));
439
440 if (GetSoundVolume() >= m_SoundWaveVolumeMax)
441 {
442 Event_OnSoundFadeInStopped();
443 SetSoundVolume(m_SoundWaveVolumeMax);
444 m_SoundWaveStarting = false;
445 }
446 }
447 else
448 {
449 SetSoundVolume(m_SoundWaveVolumeMax);
450 m_SoundWaveStarting = false;
451 }
452 }
453
454 // FadeOut
455 if (m_SoundWaveStopping)
456 {
457 if (m_SoundFadeOutDuration > 0)
458 {
459 if (m_SoundFadeOutInitVolume == 0)
460 {
461 m_SoundFadeOutInitVolume = GetSoundVolume();
462 Event_OnSoundFadeOutStarted();
463 }
464 SetSoundVolume(GetSoundVolume() - (m_SoundFadeOutInitVolume / m_SoundFadeOutDuration * time_delta));
465 }
466 else
467 SetSoundVolume(0);
468
469 if (GetSoundVolume() <= 0)
470 {
471 if (m_SoundWaveObject)
472 {
473 m_SoundWaveObject.Stop();
474 m_SoundWaveStopping = false;
475 m_SoundFadedOut = true;
476 }
477 }
478 }
479
480 // Counting timer here because loop play
481 m_SoundWaveTime += time_delta;
482 }
