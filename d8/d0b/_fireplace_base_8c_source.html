1170 bool GetCeilingHeight(out float actual_height)
1171 {
1172 vector from = this.GetPosition();
1173 vector to = this.GetPosition();
1174 from[1] = from[1] + 1.0;
1175 to[1] = to[1] + MIN_CEILING_HEIGHT;
1176 vector contactPos;
1177 vector contactDir;
1178 int contactComponent;
1179
1180 bool hit = DayZPhysics.RaycastRV(from, to, contactPos, contactDir, contactComponent, NULL, NULL, this);
1181 actual_height = vector.Distance(from, contactPos) + 1.0;
1182
1183 return hit;
1184 }
